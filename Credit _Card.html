def ans (tp,tn,fp,fn,ac):
    d = pd.Series(index =['tp','tn','fp','fn','spec','sens','ac'])
    
    spec =tn/(tn+fp)
    sens = tp/(tp+fn)
    d = pd.Series(index =['tp','tn','fp','fn','spec','sens','ac'] , data =[tp,tn,fp,fn,spec,sens,ac])
    return d
